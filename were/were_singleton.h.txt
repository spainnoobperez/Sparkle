#ifndef WERE_SINGLETON_H
#define WERE_SINGLETON_H

#include <cstdio>

class were_singleton
{
public:

    static were_singleton &instance()
    {
        static were_singleton instance;
        return instance;
    }

private:
    ~were_singleton() {fprintf(stdout, "down\n");}
    were_singleton() {fprintf(stdout, "up\n");}

public:
    were_singleton(const were_singleton &other) = delete;
    void operator=(const were_singleton &other) = delete;

public:
    void test() {fprintf(stdout, "test\n");}
};

#endif // WERE_SINGLETON_H
